Задание:
  Внесите правки в данный код. 
  Сделайте, чтобы при наличии ошибки во время восстановления объекта из JSON строки
  выводилось сообщение - "Ошибка при чтении JSON"

	let json = '{"name" : "Ivan", "age" = 20}';
	let jsonGood = '{"name" : "Ivan", "age" : 20}';
	
	try {
	  let obj = JSON.parse(json);
	  console.log(obj.name);
	  console.log(obj.age);
	}
	catch {
	  console.log(`Ошибка при чтении JSON`);
	  alert(`Ошибка при чтении JSON`);
	}
	
	try {
	  let obj = JSON.parse(jsonGood);
	  console.log(obj.name);
	  console.log(obj.age);
	}
	catch {
	  console.log(`Ошибка при чтении JSON`);
	  alert(`Ошибка при чтении JSON`);
	}
------------------------------------------------------------------------------
функция JSON.parse должна вернуть объект, но не может этого сделать 
из-за неправильных входных данных, она выбрасывает исключение (генерирует ошибку),
которая останавливает дальнейшее выполнение сценария.
-------
исключение прекращает выполнение всех последующих инструкций в блоке try 
и переносит выполнение в блок catch.
