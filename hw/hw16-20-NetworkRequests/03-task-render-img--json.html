Задание:
  Напишите сценарий, 
  который при загрузке страницы выполнит GET запрос 
  на https://jsonplaceholder.typicode.com/photos
  и выведет на странице все изображения в виде ,
  которые будут получены в ответе.
-----------------------------------------------------------
		const output = document.querySelector('.output')
		let url = "https://jsonplaceholder.typicode.com/photos";
		let promise = fetch(url);
		
		promise
		  .then(response => response.json(), error => console.error(error.message))
		  .then(json => renderImage(json));
		
		function renderImage(photos) {
		  photos.forEach(photo => {
		    let img = document.createElement("img");
		    img.src = photo.thumbnailUrl;
		    output.append(img);
		  });
		}
